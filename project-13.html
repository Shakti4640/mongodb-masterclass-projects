🔴 REDSHIFT MINDSET — The SQL Power Tools You Know

Window Functions in Redshift

-- ═══════════════════════════════════════════════════════
-- NEED 1: Window Functions (Redshift — you know these well)
-- ═══════════════════════════════════════════════════════

-- Rank customers by spending within each city
SELECT
    customer_id,
    city,
    total_spent,
    RANK() OVER (PARTITION BY city ORDER BY total_spent DESC) AS city_rank,
    DENSE_RANK() OVER (PARTITION BY city ORDER BY total_spent DESC) AS dense_rank,
    ROW_NUMBER() OVER (PARTITION BY city ORDER BY total_spent DESC) AS row_num,
    SUM(total_spent) OVER (PARTITION BY city) AS city_total,
    total_spent / SUM(total_spent) OVER (PARTITION BY city) * 100 AS pct_of_city,
    NTILE(4) OVER (PARTITION BY city ORDER BY total_spent DESC) AS quartile
FROM customer_summary
ORDER BY city, city_rank;

-- Running total of monthly revenue
SELECT
    month,
    revenue,
    SUM(revenue) OVER (ORDER BY month ROWS UNBOUNDED PRECEDING) AS running_total,
    AVG(revenue) OVER (ORDER BY month ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) AS moving_avg_3m,
    LAG(revenue, 1) OVER (ORDER BY month) AS prev_month,
    LEAD(revenue, 1) OVER (ORDER BY month) AS next_month,
    revenue - LAG(revenue, 1) OVER (ORDER BY month) AS month_over_month_change
FROM monthly_revenue
ORDER BY month;

-- HOW WINDOW FUNCTIONS WORK:
-- 1. PARTITION BY = defines the "window" (group without collapsing)
-- 2. ORDER BY = defines order within the window
-- 3. ROWS/RANGE = defines the frame (which rows to include)
-- 4. Function applied within the frame per row
-- 5. Original rows preserved (unlike GROUP BY which collapses)

JOINs in Redshift

-- ═══════════════════════════════════════════════════════
-- NEED 2: JOINs (Redshift — your bread and butter)
-- ═══════════════════════════════════════════════════════

-- Orders enriched with customer details
SELECT
    o.order_id,
    o.order_date,
    o.total_amount,
    o.status,
    c.name AS customer_name,
    c.city,
    c.loyalty_tier
FROM store.orders o
LEFT JOIN store.customers c ON o.customer_id = c.customer_id
WHERE o.order_date >= '2024-01-01'
ORDER BY o.order_date DESC;

-- Redshift JOIN mechanics:
-- DISTRIBUTION KEY matters:
--   If orders.customer_id = DISTKEY AND customers.customer_id = DISTKEY
--   → CO-LOCATED JOIN (data already on same node) → FAST
--   If different DISTKEYs:
--   → BROADCAST or REDISTRIBUTE (data movement between nodes) → SLOW
--
-- This is why DISTKEY choice is critical in Redshift
-- MongoDB has no equivalent concern ($lookup always reads from target collection)

// ═══════════════════════════════════════════════════════
// $setWindowFields — RANK, ROW_NUMBER, Running Totals, LAG/LEAD
// Added in MongoDB 5.0 — THE stage that made MongoDB competitive
// with SQL window functions
// ═══════════════════════════════════════════════════════

// ─── FIRST: Prepare customer spending summary ────────
// (We need aggregated data to apply window functions to)

print("=== Step 1: Build customer spending summary ===\n")

// This is like creating a CTE or subquery in Redshift
let customerSpending = db.orders_detailed.aggregate([
    {
        $group: {
            _id: "$customer_id",
            total_spent: { $sum: "$total_amount" },
            order_count: { $sum: 1 },
            avg_order: { $avg: "$total_amount" }
        }
    },
    // Lookup customer details to get city
    {
        $lookup: {
            from: "customers_enriched",
            localField: "_id",
            foreignField: "_id",
            as: "customer_info"
        }
    },
    { $unwind: "$customer_info" },
    {
        $project: {
            _id: 0,
            customer_id: "$_id",
            name: "$customer_info.name",
            city: "$customer_info.city",
            tier: "$customer_info.loyalty_tier",
            total_spent: { $round: ["$total_spent", 2] },
            order_count: 1,
            avg_order: { $round: ["$avg_order", 2] }
        }
    },
    { $out: "customer_spending" }  // Save to temporary collection
]).toArray()

print("Customer spending summary created")
db.customer_spending.find().forEach(printjson)


// ─── WINDOW FUNCTION 1: RANK within city ─────────────

// REDSHIFT SQL:
// SELECT customer_id, name, city, total_spent,
//        RANK() OVER (PARTITION BY city ORDER BY total_spent DESC) AS city_rank,
//        DENSE_RANK() OVER (PARTITION BY city ORDER BY total_spent DESC) AS dense_rank,
//        ROW_NUMBER() OVER (PARTITION BY city ORDER BY total_spent DESC) AS row_num
// FROM customer_spending
// ORDER BY city, city_rank;

print("\n=== Window Function: RANK within city ===\n")

db.customer_spending.aggregate([
    {
        $setWindowFields: {
            partitionBy: "$city",                    // PARTITION BY city
            sortBy: { total_spent: -1 },             // ORDER BY total_spent DESC
            output: {
                city_rank: { $rank: {} },            // RANK()
                dense_rank: { $denseRank: {} },      // DENSE_RANK()
                row_num: { $documentNumber: {} }     // ROW_NUMBER()
            }
        }
    },
    {
        $project: {
            _id: 0,
            customer_id: 1,
            name: 1,
            city: 1,
            total_spent: 1,
            city_rank: 1,
            dense_rank: 1,
            row_num: 1
        }
    },
    { $sort: { city: 1, city_rank: 1 } }
]).forEach(printjson)

// OUTPUT:
// { customer_id: "C-00007", name: "Grace Lee", city: "New York",
//   total_spent: 799.92, city_rank: 1, dense_rank: 1, row_num: 1 }
// { customer_id: "C-00001", name: "Alice Johnson", city: "New York",
//   total_spent: 2239.95, city_rank: 2, ... }
// ...

// ─── BREAKDOWN ───────────────────────────────────────
//
// $setWindowFields: {
//     partitionBy: "$city",        ← PARTITION BY city
//                                     Groups into windows without collapsing
//                                     Unlike $group: preserves individual rows
//     sortBy: { total_spent: -1 }, ← ORDER BY total_spent DESC within window
//     output: {                    ← New fields to add to each document
//         city_rank: { $rank: {} } ← Applies RANK() within the window
//     }
// }
//
// KEY DIFFERENCE FROM $group:
//     $group:           10 customers → 3 cities (collapses rows)
//     $setWindowFields: 10 customers → 10 customers (adds fields, preserves rows)
//     EXACTLY like SQL: GROUP BY collapses, OVER() preserves


// ─── WINDOW FUNCTION 2: Running Total + Partition Total ─

// REDSHIFT SQL:
// SELECT customer_id, city, total_spent,
//        SUM(total_spent) OVER (PARTITION BY city) AS city_total,
//        total_spent / SUM(total_spent) OVER (PARTITION BY city) * 100 AS pct_of_city,
//        SUM(total_spent) OVER (PARTITION BY city ORDER BY total_spent DESC
//            ROWS UNBOUNDED PRECEDING) AS running_total_in_city
// FROM customer_spending;

print("\n=== Window Function: Running total + % of city ===\n")

db.customer_spending.aggregate([
    {
        $setWindowFields: {
            partitionBy: "$city",
            sortBy: { total_spent: -1 },
            output: {
                // Total for entire partition (no window frame = entire partition)
                city_total: {
                    $sum: "$total_spent",
                    window: {
                        documents: ["unbounded", "unbounded"]   // ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
                    }
                },
                // Running total (from start of partition to current row)
                running_total: {
                    $sum: "$total_spent",
                    window: {
                        documents: ["unbounded", "current"]     // ROWS UNBOUNDED PRECEDING
                    }
                }
            }
        }
    },
    {
        $addFields: {
            pct_of_city: {
                $round: [
                    { $multiply: [{ $divide: ["$total_spent", "$city_total"] }, 100] },
                    1
                ]
            }
        }
    },
    {
        $project: {
            _id: 0, customer_id: 1, name: 1, city: 1,
            total_spent: 1, city_total: { $round: ["$city_total", 2] },
            pct_of_city: 1, running_total: { $round: ["$running_total", 2] }
        }
    },
    { $sort: { city: 1, total_spent: -1 } }
]).forEach(printjson)

// WINDOW FRAME SPECIFICATION:
//
// documents: ["unbounded", "unbounded"]  = ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
//                                          (entire partition)
// documents: ["unbounded", "current"]    = ROWS UNBOUNDED PRECEDING
//                                          (from start to current row)
// documents: [-2, "current"]             = ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
//                                          (3-row moving window)
// documents: [-1, 1]                     = ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
//                                          (3-row centered window)
// documents: ["current", "unbounded"]    = ROWS BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING


// ─── WINDOW FUNCTION 3: LAG / LEAD ($shift) ─────────

// REDSHIFT SQL:
// SELECT month, revenue,
//        LAG(revenue, 1) OVER (ORDER BY month) AS prev_month_revenue,
//        LEAD(revenue, 1) OVER (ORDER BY month) AS next_month_revenue,
//        revenue - LAG(revenue, 1) OVER (ORDER BY month) AS month_change
// FROM monthly_revenue;

// First create monthly revenue data
print("\n=== Step: Build monthly revenue ===\n")

db.orders_detailed.aggregate([
    {
        $group: {
            _id: {
                year: { $year: "$order_date" },
                month: { $month: "$order_date" }
            },
            revenue: { $sum: "$total_amount" },
            order_count: { $sum: 1 }
        }
    },
    { $sort: { "_id.year": 1, "_id.month": 1 } },
    {
        $project: {
            _id: 0,
            month_date: {
                $dateFromParts: { year: "$_id.year", month: "$_id.month", day: 1 }
            },
            month_label: {
                $concat: [
                    { $toString: "$_id.year" }, "-",
                    { $cond: {
                        if: { $lt: ["$_id.month", 10] },
                        then: { $concat: ["0", { $toString: "$_id.month" }] },
                        else: { $toString: "$_id.month" }
                    }}
                ]
            },
            revenue: { $round: ["$revenue", 2] },
            order_count: 1
        }
    },
    { $out: "monthly_revenue" }
]).toArray()

print("Monthly revenue:")
db.monthly_revenue.find().sort({ month_date: 1 }).forEach(printjson)

// Now apply LAG/LEAD
print("\n=== Window Function: LAG / LEAD ($shift) ===\n")

db.monthly_revenue.aggregate([
    {
        $setWindowFields: {
            sortBy: { month_date: 1 },               // ORDER BY month
            output: {
                prev_month_revenue: {
                    $shift: {                          // LAG equivalent
                        output: "$revenue",
                        by: -1,                        // -1 = previous row (LAG)
                        default: 0                     // default when no previous
                    }
                },
                next_month_revenue: {
                    $shift: {                          // LEAD equivalent
                        output: "$revenue",
                        by: 1,                         // +1 = next row (LEAD)
                        default: 0
                    }
                }
            }
        }
    },
    {
        $addFields: {
            month_change: {
                $round: [{ $subtract: ["$revenue", "$prev_month_revenue"] }, 2]
            },
            growth_pct: {
                $cond: {
                    if: { $eq: ["$prev_month_revenue", 0] },
                    then: null,
                    else: {
                        $round: [
                            { $multiply: [
                                { $divide: [
                                    { $subtract: ["$revenue", "$prev_month_revenue"] },
                                    "$prev_month_revenue"
                                ]},
                                100
                            ]},
                            1
                        ]
                    }
                }
            }
        }
    },
    {
        $project: {
            _id: 0, month_label: 1, revenue: 1,
            prev_month_revenue: 1, next_month_revenue: 1,
            month_change: 1, growth_pct: 1
        }
    },
    { $sort: { month_date: 1 } }
]).forEach(printjson)

// $shift MAPPING:

// ─── BASIC $lookup (simple foreign key JOIN) ─────────

// REDSHIFT SQL:
// SELECT o.*, c.name, c.city, c.loyalty_tier

            foreignField: "_id",                // customers._id

        }
    },
    // $lookup ALWAYS returns an ARRAY (even for 1:1 relationships)
    // customer: [{ _id: "C-00001", name: "Alice", ... }]
    // We need to "unwrap" it

    // Stage 2: Unwind the single-element array
    {
        $unwind: {
            path: "$customer",
            preserveNullAndEmptyArrays: true    // keep orders with no matching customer
                                                // = LEFT JOIN behavior
                                                // without this = INNER JOIN behavior
        }

    // Stage 3: Project final output
    {
        $project: {
            _id: 0,
            order_id: 1,
            order_date: { $dateToString: { format: "%Y-%m-%d", date: "$order_date" } },
            total_amount: 1,
            status: 1,
            // Flattened customer fields (like SQL JOIN result)
            customer_name: "$customer.name",
            customer_city: "$customer.city",
            customer_tier: "$customer.loyalty_tier"
        }
    },
    { $sort: { order_date: -1 } }
]).forEach(printjson)

// KEY CONCEPTS:
//
// $lookup ALWAYS produces an ARRAY result:
//   → 1 match: customer: [{ one document }]
//   → 0 matches: customer: []  (empty array = LEFT JOIN null)
//   → N matches: customer: [{ doc1 }, { doc2 }, ...]  (1-to-many)
//
// $unwind AFTER $lookup flattens:
//   → { customer: [{ doc }] } → { customer: { doc } }
//   → preserveNullAndEmptyArrays: true → keeps unmatched (LEFT JOIN)
//   → preserveNullAndEmptyArrays: false → drops unmatched (INNER JOIN)
//
// REDSHIFT COMPARISON:
//   LEFT JOIN → $lookup + $unwind(preserveNull: true)
//   INNER JOIN → $lookup + $unwind(preserveNull: false)
//   RIGHT JOIN → swap collections (lookup from the other direction)
//   FULL OUTER JOIN → not directly supported (use $unionWith + dedup)


// ─── PIPELINE $lookup (correlated subquery / complex JOIN) ──

// REDSHIFT SQL:
// SELECT c.*,
//        (SELECT COUNT(*) FROM orders o WHERE o.customer_id = c.customer_id) AS order_count,
//        (SELECT SUM(total_amount) FROM orders o WHERE o.customer_id = c.customer_id) AS total_spent

// SELECT c.*, sub.order_count, sub.total_spent
// FROM customers c
// LEFT JOIN (
//     SELECT customer_id, COUNT(*) as order_count, SUM(total_amount) as total_spent
//     FROM orders GROUP BY customer_id

db.customers_enriched.aggregate([
    {
        $lookup: {
            from: "orders_detailed",
            // Instead of localField/foreignField, use pipeline:
            let: { cust_id: "$_id" },           // variables from outer document
            pipeline: [                           // sub-pipeline (correlated subquery)
                {
                    $match: {
                        $expr: {                  // match using variables
                            $eq: ["$customer_id", "$$cust_id"]  // $$ = variable reference
                        }
                    }
                },
                {
                    $group: {
                        _id: null,
                        order_count: { $sum: 1 },

                }

            as: "order_summary"

    },
    // Unwind (may be empty array for customers with no orders)
    {
        $unwind: {
            path: "$order_summary",
            preserveNullAndEmptyArrays: true
        }
    },
    {
        $project: {
            _id: 0,
            customer_id: "$_id",
            name: 1,
            city: 1,
            tier: "$loyalty_tier",
            order_count: { $ifNull: ["$order_summary.order_count", 0] },
            total_spent: {
                $round: [{ $ifNull: ["$order_summary.total_spent", 0] }, 2]
            },
            last_order: {
                $dateToString: {
                    format: "%Y-%m-%d",
                    date: { $ifNull: ["$order_summary.last_order", null] }
                }
            }
        }
    },
    { $sort: { total_spent: -1 } }
]).forEach(printjson)

⚡ WHAT COULD BE DONE (Advanced Pipeline Optimizations)

1. INDEX ON $lookup FOREIGN FIELD
   → $lookup joins on foreignField — MUST have index on it
   → db.customers_enriched has _id index (auto) → good
   → If joining on non-_id field: CREATE INDEX on that field
   → Without index: $lookup does COLLSCAN per outer document → terrible
   → Redshift equivalent: DISTKEY co-location for fast JOINs

2. $unwind BEFORE $lookup (reduce $lookup executions)
   → If unwinding produces fewer unique join keys:
     → Unwind first → fewer distinct customer_ids → fewer lookups
   → If unwinding multiplies documents:
     → Lookup first → fewer documents to unwind → less work
   → MEASURE both orders with explain()

3. $setWindowFields WITH INDEX-SUPPORTED $sort
   → If sortBy matches an index → faster sorting within windows
   → If partitionBy matches first field of compound index → faster grouping

4. AVOID $lookup WHEN EMBEDDING IS POSSIBLE
   → $lookup = JOIN = expensive (especially pipeline $lookup)
   → If you always need customer name with orders → embed it
   → { customer_id: "C-00001", customer_name: "Alice" } ← denormalize
   → Eliminates $lookup entirely → massive speedup
   → THIS IS THE MongoDB WAY (Project 14 covers deeply)

5. USE $merge TO MATERIALIZE COMPLEX PIPELINES
   → Pipeline with $unwind + $lookup + $group + $setWindowFields = slow
   → Run once → $merge results into a summary collection
   → Dashboard queries read from summary → instant
   → Refresh periodically (like Redshift MATERIALIZED VIEW)

❌ WHAT SHOULD NOT DO

DO NOT:
    ❌ Use $lookup as freely as Redshift JOINs
       → Redshift JOINs: optimized with DISTKEY co-location, MPP parallel
       → MongoDB $lookup: reads from target collection per document
       → $lookup is EXPENSIVE compared to SQL JOIN
       → Design documents to minimize $lookup need
       → THIS IS THE BIGGEST MINDSET SHIFT for Redshift developers
    
    ❌ Forget preserveNullAndEmptyArrays on $unwind after $lookup
       → $lookup returns empty array [] for no match
       → $unwind drops empty arrays by default
       → Result: LEFT JOIN silently becomes INNER JOIN
       → ALWAYS use: { $unwind: { path: "$field", preserveNullAndEmptyArrays: true } }
    
    ❌ Use pipeline $lookup when basic $lookup works
       → Basic: localField/foreignField → simple, fast
       → Pipeline: let/pipeline → flexible but slower (runs per document)
       → Use pipeline only when you need filtering/aggregation in the join
    
    ❌ $unwind large arrays without $match first
       → Order with 1000 items → $unwind produces 1000 documents
       → 1M orders × 10 items = 10M documents after $unwind
       → Filter with $match BEFORE $unwind to reduce volume
    
    ❌ Chain multiple $lookups without considering alternatives
       → $lookup A → $lookup B → $lookup C = 3 collection reads per document
       → Consider: embed A and B data, only $lookup C
       → Or: pre-join into a materialized collection with $merge
    
    ❌ Use $setWindowFields without understanding memory impact
       → Window functions buffer partition data in memory
       → Large partitions (1M docs in one city) → memory pressure
       → Use allowDiskUse: true for large windows
       → Or filter with $match to reduce partition sizes

⚙️ HOW IT ONLY WORKS INTERNALLY

$setWindowFields INTERNALS:
    → Sorts documents by sortBy within each partition
    → Maintains a sliding window frame in memory
    → Applies window function to each document's frame
    → STREAMING for rank/documentNumber (only needs sort order)
    → BUFFERING for SUM/AVG with frame (needs to hold frame in memory)
    → Large partitions → more memory → potential disk spill
    
    Redshift comparison:
    → Redshift window functions execute on compute slices
    → Each slice processes its partition in parallel (MPP)
    → Data already sorted by sort key → window functions are cheap
    → MongoDB: single-threaded, must sort first → more expensive

$lookup INTERNALS:
    → For each document in the outer collection:
      1. Extract localField value
      2. Query the "from" collection with foreignField = localField
      3. If index exists on foreignField → IXSCAN (fast)
      4. If no index → COLLSCAN (catastrophically slow)
      5. Append results as array field
    
    → Basic $lookup: one indexed query per outer document
    → Pipeline $lookup: runs full sub-pipeline per outer document
    
    → Performance: O(n × log m) with index, O(n × m) without
      where n = outer docs, m = inner collection size
    
    Redshift comparison:
    → Hash JOIN: build hash table on smaller table → probe with larger
    → Merge JOIN: both tables sorted → merge in order
    → Broadcast JOIN: small table sent to all nodes
    → All much more efficient than MongoDB's nested loop approach
    → THIS IS WHY: denormalize in MongoDB, normalize in Redshift

$unwind INTERNALS:
    → For each document:
      1. Read the array field
      2. For each element: create a copy of the document
      3. Replace array with single element
      4. Pass each copy to next stage
    → Document with N-element array → N documents out
    → Memory: must hold the document + create N copies
    → Large arrays: significant memory amplification
    
    Redshift comparison:
    → LATERAL FLATTEN: similar concept for SUPER type
    → Normalized tables: no flatten needed (already one row per element)
    → MongoDB $unwind = the cost of storing arrays instead of normalizing

⚠️ COMMON PITFALLS AND FAILURE MODES

MISTAKES REDSHIFT DEVELOPERS MAKE — PROJECT 13 SPECIFIC:

PITFALL 1: "I'll use $lookup like I use JOIN — everywhere"
    ❌ Building pipelines with 3-4 $lookup stages like SQL with multiple JOINs
    ✅ Denormalize frequently-joined data into the document
    WHY: Redshift JOINs are optimized with DISTKEY, hash join, merge join
         MongoDB $lookup is a nested loop join (one query per outer doc)
         3 $lookups on 10K docs = 30K sub-queries → SLOW
         Embed what you frequently join → zero lookups needed
    RULE: If you $lookup the same collections repeatedly →
          you should have embedded that data instead
    THIS IS THE #1 LESSON: MongoDB is NOT Redshift with different syntax
         It's a fundamentally different data model

PITFALL 2: "Forgetting preserveNullAndEmptyArrays on $unwind after $lookup"
    ❌ $lookup: { ... as: "customer" }
       $unwind: "$customer"                    ← DROPS orders with no customer match!
    ✅ $unwind: { path: "$customer", preserveNullAndEmptyArrays: true }
    WHY: $lookup returns empty array [] for no match (LEFT JOIN behavior)
         $unwind by default DROPS documents with empty arrays
         Without preserveNull: LEFT JOIN silently becomes INNER JOIN
         You lose data without any error or warning
    IMPACT: Silent data loss — the most dangerous bug type
    FREQUENCY: EVERY Redshift developer hits this at least once

PITFALL 3: "No index on $lookup foreignField"
    ❌ $lookup: { from: "customers", foreignField: "customer_id", ... }
       where customers has NO index on customer_id
    ✅ db.customers.createIndex({ customer_id: 1 })
    WHY: Without index: COLLSCAN on customers collection PER outer document
         10K orders × COLLSCAN on 50K customers = 500M document comparisons
         With index: 10K orders × IXSCAN = 10K index lookups → 1000x faster
    REDSHIFT EQUIVALENT: This is like having no DISTKEY and forcing
         redistribute join on every query
    RULE: ALWAYS verify index exists on $lookup foreignField

PITFALL 4: "$setWindowFields without understanding partition size"
    ❌ $setWindowFields with partitionBy on low-cardinality field
       → partition of 5M documents → all buffered in memory
    ✅ Filter with $match first to reduce partition sizes
       → Or ensure partitions are reasonably sized (<100K docs)
    WHY: Window functions buffer partition data in memory
         Redshift handles this with MPP (distributed across nodes)
         MongoDB single-threaded → large partitions = memory pressure
    FIX: allowDiskUse: true (slower but won't crash)
         Or $match to reduce data before $setWindowFields

PITFALL 5: "Using $unwind on huge arrays without filtering first"
    ❌ Collection with 1M docs, each with 100-element array
       → $unwind produces 100M documents → pipeline overwhelmed
    ✅ $match first to filter to relevant documents
       → Or $project to reduce fields before $unwind
       → Or use $filter (array filter operator) before $unwind
    WHY: $unwind is a document MULTIPLIER
         10 items per order × 1M orders = 10M documents flowing through pipeline
         Each subsequent stage processes 10M docs instead of 1M
    EXAMPLE:
         // GOOD: filter first, then unwind
         { $match: { customer_id: "C-00001" } },    // 5 orders
         { $unwind: "$items" }                       // 15 items
         
         // BAD: unwind first, then filter
         { $unwind: "$items" },                      // 5M items (from all orders!)
         { $match: { customer_id: "C-00001" } }      // filter to 15 items
         
         // Same result, 333,000x different work

PITFALL 6: "Expecting $shift (LAG/LEAD) without sortBy"
    ❌ $setWindowFields without sortBy using $shift
       → Document order is undefined → $shift returns unpredictable values
    ✅ ALWAYS specify sortBy when using $shift, $rank, or frame-based functions
    WHY: LAG/LEAD depend on row ORDER within the window
         Without sortBy: order is random → LAG returns random previous document
         Redshift: ORDER BY is required in OVER clause → SQL enforces this
         MongoDB: doesn't error without sortBy → silently wrong results

PITFALL 7: "Using pipeline $lookup when basic $lookup works"
    ❌ $lookup with let + pipeline for simple equality join
       pipeline: [{ $match: { $expr: { $eq: ["$_id", "$$cust_id"] } } }]
    ✅ Basic $lookup: localField/foreignField (simpler, faster)
       $lookup: { from: "customers", localField: "customer_id",
                  foreignField: "_id", as: "customer" }
    WHY: Basic $lookup is internally optimized (can use index directly)
         Pipeline $lookup runs a full sub-pipeline per outer document
         Basic is ~2-5x faster for simple equality joins
    WHEN TO USE PIPELINE: Only when you need to filter, aggregate,
         or transform within the join (correlated subquery scenarios)

PITFALL 8: "Assuming $lookup result is a single document"
    ❌ After $lookup: accessing "$customer.name" directly
       → ERROR: customer is an ARRAY, not an object
    ✅ After $lookup: $unwind first, THEN access "$customer.name"
       → Or use $arrayElemAt: ["$customer", 0] for single match
    WHY: $lookup ALWAYS returns an array, even for 1:1 relationships
         [{ name: "Alice" }] ← array with one element
         Must unwrap: $unwind or $arrayElemAt
         Redshift JOIN returns flat columns — no array wrapping
         This is a fundamental MongoDB difference

📈 SCALABILITY CONSIDERATIONS

ADVANCED PIPELINE SCALABILITY:

$setWindowFields SCALING:
    → Memory: proportional to partition size
    → 100 docs per partition: trivial (in-memory)
    → 10K docs per partition: manageable
    → 1M docs per partition: needs allowDiskUse
    → Redshift: MPP distributes partitions across nodes → handles larger partitions
    
    OPTIMIZATION:
    → Small partitions: partition by high-cardinality field
    → $match before $setWindowFields: reduce total data
    → allowDiskUse: true for safety

$lookup SCALING:
    → O(n × log m) with index: n=outer docs, m=inner collection
    → 100 outer docs × 50K inner (indexed): ~50ms
    → 10K outer docs × 50K inner (indexed): ~5 seconds
    → 1M outer docs × 50K inner (indexed): ~500 seconds (8+ minutes!)
    → 1M outer docs × 50K inner (NO index): HOURS
    
    COMPARISON TO REDSHIFT:
    → Redshift hash join: O(n + m) — build hash table + probe
    → 1M × 50K: ~3-5 seconds (hash join, MPP parallel)
    → MongoDB: ~500 seconds for same operation
    → MongoDB is 100x SLOWER for large full-collection JOINs
    
    WHEN MONGODB $lookup IS ACCEPTABLE:
    → $match narrows outer to <1000 docs before $lookup
    → Inner collection has index on foreignField
    → Total $lookup executions < 10K
    
    WHEN TO USE REDSHIFT INSTEAD:
    → Joining millions of rows from both tables
    → Multiple JOINs in one query
    → Regular analytical reports requiring JOINs

$unwind SCALING:
    → Document count multiplied by average array length
    → 1M docs × 5 items avg = 5M docs after $unwind
    → 1M docs × 100 items avg = 100M docs after $unwind
    → Memory: each unwound doc is a full copy (minus array)
    → Disk: allowDiskUse for large unwind results
    
    OPTIMIZATION:
    → $match BEFORE $unwind (reduce docs entering unwind)
    → $project BEFORE $unwind (reduce doc size per copy)
    → $unwind AFTER $lookup (not before, unless strategically better)

COMBINED PIPELINE SCALING:
    $match → $unwind → $lookup → $group → $setWindowFields

    With proper $match first (100 docs):
        → $unwind: 100 × 5 = 500 docs
        → $lookup: 500 × indexed query = ~250ms
        → $group: 500 docs → 20 groups = instant
        → $setWindowFields: 20 docs → instant
        → TOTAL: ~300ms ✅

    Without $match (1M docs):
        → $unwind: 1M × 5 = 5M docs
        → $lookup: 5M × indexed query = HOURS
        → $group: 5M docs → ? groups = minutes
        → $setWindowFields: ? docs → depends
        → TOTAL: HOURS ❌

    THE LESSON: $match first is not optional — it's survival

📐 DATA MODELING COMPARISON

HOW $lookup NEEDS SHOULD INFLUENCE DOCUMENT DESIGN:

ANTI-PATTERN (Redshift thinking in MongoDB):
    orders:     { customer_id: "C-001", ... }
    customers:  { _id: "C-001", name: "Alice", city: "NY" }
    products:   { _id: "LAPTOP-001", name: "Gaming Laptop" }
    
    Every query needs 2-3 $lookups → slow
    This is normalized Redshift design → wrong for MongoDB

GOOD PATTERN (MongoDB thinking):
    orders: {
        customer_id: "C-001",

        customer_city: "New York",         ← embedded
        items: [
            { sku: "LAPTOP-001", name: "Gaming Laptop", price: 1299.99 }
        ]                                  ← embedded
    }
    
    Zero $lookups needed for most queries → fast
    Customer name duplicated across orders → accepted trade-off
    Product name duplicated across orders → accepted trade-off

def run_pipeline(collection, pipeline, label, max_results=15):
    """Run pipeline with timing."""
    print(f"\n  {'='*60}")
    print(f"  {label}")
    print(f"  {'='*60}")
    
    start = time.time()
    results = list(collection.aggregate(pipeline, allowDiskUse=True))
    elapsed = (time.time() - start) * 1000
    
    print(f"  Results: {len(results)} | Time: {elapsed:.1f}ms")
    print(f"  {'─'*60}")
    
    for doc in results[:max_results]:
        clean = {}
        for k, v in doc.items():
            if isinstance(v, float):
                clean[k] = round(v, 2)
            elif isinstance(v, list) and len(v) > 5:
                clean[k] = f"[{len(v)} items]"
            elif isinstance(v, datetime):
                clean[k] = v.strftime("%Y-%m-%d")
            else:
                clean[k] = v
        pprint.pprint(clean)
    
    if len(results) > max_results:
        print(f"  ... ({len(results) - max_results} more)")
    
    return results

    {"$sort": {"order_date": -1}}
], "LOOKUP: Orders LEFT JOIN Customers")


# ═══════════════════════════════════════════════════════
# 5. $lookup: Pipeline (correlated subquery)
# ═══════════════════════════════════════════════════════

run_pipeline(db.customers_enriched, [
    {
        "$lookup": {
            "from": "orders_detailed",
            "let": {"cust_id": "$_id"},
            "pipeline": [
                {"$match": {"$expr": {"$eq": ["$customer_id", "$$cust_id"]}}},
                {
                    "$group": {
                        "_id": None,
                        "order_count": {"$sum": 1},
                        "total_spent": {"$sum": "$total_amount"},
                        "last_order": {"$max": "$order_date"}
                    }
                },
                {"$project": {"_id": 0}}
            ],
            "as": "stats"
        }
    },
    {"$unwind": {"path": "$stats", "preserveNullAndEmptyArrays": True}},
    {
        "$project": {
            "_id": 0,
            "customer_id": "$_id",
            "name": 1,
            "city": 1,
            "tier": "$loyalty_tier",
            "orders": {"$ifNull": ["$stats.order_count", 0]},
            "total_spent": {"$round": [{"$ifNull": ["$stats.total_spent", 0]}, 2]},
            "last_order": {
                "$cond": {
                    "if": {"$eq": [{"$type": "$stats.last_order"}, "missing"]},
                    "then": "never",
                    "else": {"$dateToString": {"format": "%Y-%m-%d", "date": "$stats.last_order"}}
                }
            }
        }
    },
    {"$sort": {"total_spent": -1}}
], "LOOKUP: Customers with order summary (correlated subquery)")


# ═══════════════════════════════════════════════════════
# 6. $unwind: Flatten items + Group by product
# ═══════════════════════════════════════════════════════

run_pipeline(db.orders_detailed, [
    {"$unwind": "$items"},
    {
        "$group": {
            "_id": {"sku": "$items.sku", "name": "$items.name"},
            "total_qty": {"$sum": "$items.qty"},
            "total_revenue": {"$sum": {"$multiply": ["$items.qty", "$items.price"]}},
            "order_count": {"$addToSet": "$order_id"}
        }
    },
    {
        "$project": {
            "_id": 0,
            "sku": "$_id.sku",
            "product": "$_id.name",
            "total_qty": 1,
            "total_revenue": {"$round": ["$total_revenue", 2]},
            "unique_orders": {"$size": "$order_count"}
        }
    },
    {"$sort": {"total_revenue": -1}}
], "UNWIND + GROUP: Revenue per product (SKU)")


# ═══════════════════════════════════════════════════════
# 7. GRAND COMBINATION: unwind + lookup + group + window
# ═══════════════════════════════════════════════════════

run_pipeline(db.orders_detailed, [
    # Flatten items
    {"$unwind": "$items"},
    # Join with customers
    {
        "$lookup": {
            "from": "customers_enriched",
            "localField": "customer_id",
            "foreignField": "_id",
            "as": "cust"
        }
    },
    {"$unwind": "$cust"},
    # Group by city + product
    {
        "$group": {
            "_id": {"city": "$cust.city", "sku": "$items.sku", "product": "$items.name"},
            "revenue": {"$sum": {"$multiply": ["$items.qty", "$items.price"]}},
            "qty": {"$sum": "$items.qty"}
        }
    },
    # Rank within city
    {
        "$setWindowFields": {
            "partitionBy": "$_id.city",
            "sortBy": {"revenue": -1},
            "output": {
                "city_rank": {"$rank": {}},
                "city_total": {
                    "$sum": "$revenue",
                    "window": {"documents": ["unbounded", "unbounded"]}
                }
            }
        }
    },
    # Format output
    {
        "$project": {
            "_id": 0,
            "city": "$_id.city",
            "product": "$_id.product",
            "qty": 1,
            "revenue": {"$round": ["$revenue", 2]},
            "city_rank": 1,
            "pct": {
                "$round": [
                    {"$multiply": [{"$divide": ["$revenue", "$city_total"]}, 100]}, 1
                ]
            }
        }
    },
    {"$sort": {"city": 1, "city_rank": 1}}
], "GRAND COMBINATION: Product revenue ranked by city")
