    Need 1: SEARCH — Users type "comfortable running shoes lightweight"
        → Find products matching ANY of those words in name/description
        → Rank results by relevance
        → Handle typos and stemming ("running" matches "run", "runner")
    
    Need 2: GEO — "Find stores within 5km of my current location"
        → Query by proximity to a GPS coordinate

-- ═══════════════════════════════════════════════════════
-- NEED 1: Full-Text Search in Redshift
-- ═══════════════════════════════════════════════════════

-- Your only options:
-- OPTION A: LIKE with wildcards (terrible performance)
SELECT * FROM products
WHERE description LIKE '%comfortable%'
  AND description LIKE '%running%'
  AND description LIKE '%lightweight%';
-- Full table scan EVERY time
-- No relevance ranking
-- No stemming ("running" doesn't match "run")
-- No word boundary awareness ("running" matches "unrunning")
-- Gets slower linearly with data size

-- OPTION B: Use SIMILAR TO or regex (still terrible)
SELECT * FROM products
WHERE description SIMILAR TO '%(comfortable|running|lightweight)%';
-- Same full scan problem
-- Slightly more flexible patterns
-- Still no ranking, no stemming

-- OPTION C: Integrate Amazon OpenSearch (external service)
-- Set up OpenSearch domain
-- Build ETL pipeline: Redshift → S3 → Lambda → OpenSearch
-- Query OpenSearch for text search
-- Join results back to Redshift for other data
-- COMPLEXITY: massive — separate infrastructure, sync pipeline, latency

-- ═══════════════════════════════════════════════════════
-- NEED 2: Geospatial Queries in Redshift
-- ═══════════════════════════════════════════════════════

-- OPTION A: GEOMETRY type with spatial functions
SELECT store_id, store_name,
    ST_DistanceSphere(
        ST_Point(longitude, latitude),
        ST_Point(-73.9857, 40.7484)
    ) AS distance_meters
FROM stores
WHERE ST_DistanceSphere(
    ST_Point(longitude, latitude),
    ST_Point(-73.9857, 40.7484)
) < 5000
ORDER BY distance_meters;
-- NO spatial index → full table scan
-- Computes distance for EVERY row → expensive
-- 1M stores → computes 1M distance calculations

WHERE JSON_ARRAY_CONTAINS(tags, 'organic');

INDEX TYPE 1: Text Index — Full-Text Search

// ═══════════════════════════════════════════════════════
// TEXT INDEX — Replaces LIKE '%keyword%' and external OpenSearch
// ═══════════════════════════════════════════════════════

// WHAT IT IS:
//   → Inverted index on string fields
//   → Tokenizes text into words
//   → Stems words: "running" → "run", "comfortable" → "comfort"
//   → Removes stop words: "the", "and", "is", "a"
//   → Supports relevance scoring
//   → Multiple languages supported

// RULE: Only ONE text index per collection
// (Can include multiple fields in that one text index)

// ─── CREATE TEXT INDEX ───────────────────────────────
db.products_full.createIndex(
    {
        name: "text",              // include product name
        description: "text",       // include description
        "tags": "text"             // include tags array (strings)
    },
    {
        name: "idx_text_search",
        weights: {                  // relevance weighting
            name: 10,              // name matches worth 10x
            tags: 5,               // tag matches worth 5x
            description: 1         // description matches worth 1x
        },
        default_language: "english"  // stemming language
    }
)

// ─── BASIC TEXT SEARCH ───────────────────────────────
// Find products matching "comfortable running"
print("=== Text search: 'comfortable running' ===")
db.products_full.find(
    { $text: { $search: "comfortable running" } },
    { name: 1, score: { $meta: "textScore" }, _id: 0 }
).sort(
    { score: { $meta: "textScore" } }    // sort by relevance
).forEach(printjson)

// OUTPUT (ranked by relevance):
// { name: "Ultra Comfort Running Shoes", score: 7.5 }     ← highest (name match)
// { name: "Organic Cotton T-Shirt", score: 2.1 }          ← "comfortable" in desc
// { name: "Wireless Noise-Canceling Headphones", score: 1.8 } ← "comfortable" in desc

// HOW SCORING WORKS:
// "comfortable" matches:
//   → Shoes: in name (weight 10) + tags (weight 5) + description (weight 1)
//   → T-Shirt: in tags (weight 5) + description (weight 1)
//   → Headphones: in tags (weight 5) + description (weight 1)
// "running" matches:
//   → Shoes: in name (weight 10) + tags (weight 5) + description (weight 1)
// Shoes win with highest combined score

// ─── PHRASE SEARCH ───────────────────────────────────

// 2DSPHERE INDEX — Built-in geospatial with INDEX support
// (Introduced in Project 8 types, now deep dive on index)
// ═══════════════════════════════════════════════════════

// From Project 8: we stored GeoJSON and queried with $near
// Now: understanding the INDEX that makes it fast

// ─── CREATE GEOSPATIAL INDEX ─────────────────────────
db.stores.createIndex(
    { location: "2dsphere" },      // special index type string
    { name: "idx_geo_location" }
)

// WHAT THIS BUILDS:
// → NOT a B-tree (unlike regular indexes)
// → S2 cell-based spatial index
// → Divides Earth's surface into hierarchical cells
// → Each cell has a unique ID
// → Documents mapped to cells based on coordinates
// → Queries map search area to cells → find matching documents
// → O(log n) for proximity queries (not full scan!)

// ─── $near — Find nearest stores ─────────────────────
print("=== Stores near Times Square (10km) ===")
db.stores.find({
    location: {
        $near: {
            $geometry: {
                type: "Point",
                coordinates: [-73.9857, 40.7484]    // Times Square
            },
            $maxDistance: 10000      // 10km in meters
        }
    }
}).forEach(s => print(`  ${s.name} (${s.city})`))
// Results sorted by distance (nearest first):
// Manhattan Flagship (0km)
// Brooklyn Store (~5km)
// Queens Outlet (~8km)
// Jersey City Store (~4km)

// ─── $geoWithin — Find stores in polygon ─────────────
print("\n=== Stores within Manhattan boundary ===")
db.stores.find({
    location: {
        $geoWithin: {
            $geometry: {
                type: "Polygon",
                coordinates: [[
                    [-74.02, 40.70],    // SW
                    [-73.93, 40.70],    // SE
                    [-73.93, 40.80],    // NE
                    [-74.02, 40.80],    // NW
                    [-74.02, 40.70]     // close polygon
                ]]
            }
        }
    }
}).forEach(s => print(`  ${s.name}`))
// Returns only stores within Manhattan polygon

// ─── $geoNear in aggregation (distance in results) ──
print("\n=== Stores with exact distance ===")
db.stores.aggregate([
    {
        $geoNear: {
            near: { type: "Point", coordinates: [-73.9857, 40.7484] },
            distanceField: "distance_meters",

        $project: {
            name: 1,
            city: 1,

]).forEach(printjson)

INDEX TYPE 3: Hashed Index — Equality-Only, Uniform Distribution

// ═══════════════════════════════════════════════════════
// HASHED INDEX — For equality queries and shard key distribution
// ═══════════════════════════════════════════════════════

// WHAT IT IS:
//   → Hashes the field value → stores hash in B-tree
//   → Equality queries: hash the search value → find in tree
//   → CANNOT support range queries ($gt, $lt, $gte, $lte)
//   → CANNOT support sort operations
//   → Uniform distribution of index entries (good for sharding)

// WHY USE IT?
//   → Primary use: SHARD KEY (Project 21-22)
//   → Monotonically increasing values (timestamps, ObjectIds)
//     distribute unevenly with range index (all new data → one shard)
//   → Hashed index distributes evenly across shards

// ─── CREATE HASHED INDEX ─────────────────────────────
db.products_full.createIndex(

INDEX TYPE 5: Partial Index — Index Only What Matters

// ═══════════════════════════════════════════════════════
// PARTIAL INDEX — Index a subset of documents
// ═══════════════════════════════════════════════════════

// WHAT IT IS:
//   → B-tree index that only includes documents matching a filter
//   → Smaller index → less RAM → faster queries
//   → Perfect when most queries target a subset of data

// SCENARIO: 80% of products are archived, 20% are active
// Only active products are searched by users
// Why index 800K archived products that nobody queries?

// ─── CREATE PARTIAL INDEX ────────────────────────────
db.products_full.createIndex(
    { category: 1, price: -1 },
    {
        name: "idx_partial_active_cat_price",
        partialFilterExpression: {
            status: "active"            // ONLY index active products
        }
    }
)

// INDEX SIZE COMPARISON (conceptual):
// Full index on 1M products: ~50MB
// Partial index on 200K active products: ~10MB
// 80% RAM savings!

// ─── QUERY THAT USES PARTIAL INDEX ───────────────────
print("=== Partial: query matching filter expression ===")
let eP1 = db.products_full.find({
    status: "active",              // MUST include partial filter condition
    category: "Electronics",
    price: { $gt: 50 }
}).explain("executionStats")
print(`Stage: ${eP1.queryPlanner.winningPlan.inputStage.stage}`)
print(`Index: ${eP1.queryPlanner.winningPlan.inputStage.indexName}`)
// Uses idx_partial_active_cat_price ✅

// ─── QUERY THAT CANNOT USE PARTIAL INDEX ─────────────
print("\n=== Partial: query WITHOUT filter match ===")
let eP2 = db.products_full.find({
    // status NOT specified — might include archived products
    category: "Electronics"
}).explain("executionStats")
// CANNOT use partial index — query might need archived products
// Falls back to COLLSCAN or another index

// CRITICAL RULE:
// Query MUST include the partialFilterExpression conditions
// to be eligible for the partial index
// If query could return documents NOT in the partial index → can't use it

// ─── PARTIAL + UNIQUE ────────────────────────────────
// Unique email only among active users (allow duplicate email in archived)
// db.users.createIndex(
//     { email: 1 },
//     {
//         unique: true,
//         partialFilterExpression: { status: "active" }
//     }
// )
// Active user: email must be unique
// Archived user: email can duplicate (deactivated accounts)
// EXTREMELY useful pattern — no Redshift equivalent

// REDSHIFT COMPARISON:
// No equivalent
// → Sort key applies to ALL rows
// → Cannot selectively sort/index a subset
// → Workaround: separate tables (active_products, archived_products)
// → Must manage data movement between tables
// → MongoDB: one collection, partial index = elegant solution

INDEX TYPE 6: Sparse Index — Skip Missing Fields

// ═══════════════════════════════════════════════════════
// SPARSE INDEX — Only index documents that HAVE the field
// ═══════════════════════════════════════════════════════

// WHAT IT IS:
//   → Skips documents where the indexed field is MISSING (not present)
//   → Documents with field = null ARE included
//   → Smaller index when many documents lack the field

// CONNECTS TO Project 8: null vs missing vs value
// → Sparse index leverages the "missing" state
// → Redshift has no "missing" state → no sparse index concept

// SCENARIO: Only some products have an isbn field (books)
// Why index isbn for products that don't have it?

db.products_full.createIndex(
    { "specs.blade_material": 1 },
    {
        name: "idx_sparse_blade",
        sparse: true                   // only index docs WITH this field
    }
)

// Only 1 document has specs.blade_material (Chef Knife Set)
// Sparse index contains only 1 entry instead of 7
// For a real collection: 1M products, 10K kitchen items with blade_material
// → Sparse index: 10K entries vs full index: 1M entries (99% smaller!)

// ─── QUERY USING SPARSE INDEX ────────────────────────
print("=== Sparse: query on sparse field ===")
db.products_full.find(
    { "specs.blade_material": "Japanese stainless steel" },
    { name: 1, "specs.blade_material": 1, _id: 0 }
).forEach(printjson)
// Uses sparse index → only checks documents that HAVE blade_material

// ─── SPARSE vs PARTIAL ──────────────────────────────
// SPARSE:
//   → Skip documents where field is MISSING
//   → Based on field EXISTENCE
//   → Simpler — just { sparse: true }
//
// PARTIAL:
//   → Skip documents not matching a FILTER EXPRESSION
//   → Based on arbitrary condition
//   → More flexible — any filter: { status: "active" }, { price: { $gt: 0 } }
//
// PARTIAL SUPERSEDES SPARSE in most cases (added MongoDB 3.2)
// Use partial when you need complex conditions
// Use sparse for simple "field exists" filtering

// REDSHIFT COMPARISON:
// No equivalent — every row has every column
// NULL values in Redshift still occupy column position
// MongoDB sparse index = skip rows that don't have the column at all

INDEX TYPE 7: Multikey Index — Index Into Arrays

// ═══════════════════════════════════════════════════════
// MULTIKEY INDEX — Index each element of an array
// THE most MongoDB-specific index type
// ═══════════════════════════════════════════════════════

// WHAT IT IS:
//   → When you create an index on a field that contains arrays
//   → MongoDB automatically creates ONE index entry PER array element
//   → Document with tags: ["a", "b", "c"] → 3 index entries
//   → Query { tags: "b" } → finds the index entry for "b" → finds document
//   → Like having a separate row per tag in a normalized table — but automatic

// ─── CREATE MULTIKEY INDEX ON TAGS ───────────────────
db.products_full.createIndex(
    { tags: 1 },
    { name: "idx_multikey_tags" }
)

// WHAT MONGODB BUILT:
// Product "Running Shoes" has tags: ["running", "sports", "comfortable", "lightweight", "breathable"]
// Index entries created:
//   "breathable"   → docPtr_shoes
//   "comfortable"  → docPtr_shoes, docPtr_tshirt, docPtr_headphones
//   "lightweight"  → docPtr_shoes, docPtr_yoga
//   "running"      → docPtr_shoes
//   "sports"       → docPtr_shoes
//   ... (entries for ALL tags across ALL documents)

// MongoDB AUTOMATICALLY detects the field is an array
// You don't specify "multikey" — it just happens
// isMultiKey: true appears in explain() output

// ─── QUERY: Find by single tag ───────────────────────
print("=== Multikey: find products with tag 'comfortable' ===")
db.products_full.find(

✅ WHAT SHOULD DO

SHOULD DO:
    ✅ Use TEXT index instead of regex for user-facing search
       → Text index: stemming, ranking, stop words, FAST
       → Regex: no stemming, no ranking, often COLLSCAN
       → Exception: regex for pattern matching (SKU codes, email domains)
    
    ✅ Use 2dsphere for ANY location-based queries
       → Always store as GeoJSON format (Project 8)
       → Always create 2dsphere index on location fields
       → Enables $near, $geoWithin, $geoIntersects
    
    ✅ Use PARTIAL indexes on large collections with clear subsets
       → Active/inactive, published/draft, current/archived
       → Dramatic RAM savings (index only the queried subset)
       → ALWAYS include partial filter in your queries
    
    ✅ Use MULTIKEY indexes for array fields you query
       → Tags, categories, sizes, colors — any array you filter on
       → Automatic: just create normal index on array field
       → Remember: max ONE array field per compound index
    
    ✅ Use WILDCARD indexes for polymorphic subdocuments
       → Product specs, user preferences, sensor metadata
       → Fields vary per document → wildcard handles all
       → Scope with path: { "specs.$**": 1 } not { "$**": 1 }
    
    ✅ Use HIDDEN indexes to test impact before dropping
       → db.coll.hideIndex("idx_name") → index invisible to planner
       → Monitor query performance → if fine, drop it
       → If performance degrades → unhide it
       → Safe way to clean up unused indexes

❌ WHAT SHOULD NOT DO

DO NOT:
    ❌ Use $regex for user-facing search when text index exists
       → find({ name: /running/i }) → COLLSCAN or inefficient index use
       → find({ $text: { $search: "running" } }) → TEXT index → fast + ranked
       → Regex is for pattern matching, not search
    
    ❌ Create more than ONE text index per collection
       → MongoDB allows only ONE text index per collection
       → Include all searchable fields in that one index
       → { name: "text", description: "text", tags: "text" }
    
    ❌ Use full wildcard { "$**": 1 } on large collections
       → Indexes EVERY field in EVERY document
       → Massive storage, massive write overhead
       → Use scoped wildcard: { "specs.$**": 1 }
       → Or use compound/single indexes for known fields
    
    ❌ Forget partialFilterExpression in queries using partial indexes
       → Partial index on { status: "active" }
       → Query WITHOUT status filter → index NOT used → COLLSCAN
       → ALWAYS include the partial filter condition in your query
    
    ❌ Create compound index with TWO array fields
       → { tags: 1, sizes: 1 } where both are arrays → ERROR
       → MongoDB cannot create Cartesian product of two arrays
       → Solution: two separate multikey indexes
       → Or embed one array differently
    
    ❌ Use hashed index when you need range queries
       → Hashed: equality ONLY ({ field: exactValue })
       → Range ($gt, $lt, $gte, $lte): NOT supported
       → Sort: NOT supported
       → Use hashed only for shard keys needing uniform distribution
    
    ❌ Think "MongoDB text search = Elasticsearch replacement"
       → MongoDB text index: basic search, good for simple needs
       → Atlas Search (Project 29): Lucene-based, full Elasticsearch-level power
       → For complex search (fuzzy, autocomplete, facets) → Atlas Search
       → For simple keyword search → text index is sufficient

⚙️ HOW IT ONLY WORKS INTERNALLY — Each Index Type

TEXT INDEX INTERNALS:
    → NOT a B-tree on raw text
    → Inverted index: word → [document pointers]
    → Process: tokenize → stem → remove stop words → store
    → "comfortable running shoes" → ["comfort", "run", "shoe"]
    → Each stem word → B-tree entry → document pointer list
    → Score calculation: TF-IDF variant with field weights
    → Redshift: no equivalent data structure

2DSPHERE INDEX INTERNALS:
    → NOT a B-tree on coordinates
    → S2 geometry library (Google's spatial indexing)
    → Earth's surface divided into hierarchical cells
    → Each cell has a 64-bit cell ID
    → Document's location → mapped to cell ID(s)
    → Index: B-tree on cell IDs
    → Proximity query: target point → nearby cell IDs → index lookup
    → Much more efficient than computing distance for every document
    → Redshift ST_Distance: computes for every row (no spatial index)

HASHED INDEX INTERNALS:
    → B-tree on HASH of field value (not raw value)
    → hash("Electronics") → 0x7a3f... → stored in B-tree
    → Equality: hash(search_value) → find in B-tree → exact match
    → Range: hash values have no order relationship → can't range scan
    → Uniform distribution: hash function spreads values evenly
    → Critical for sharding: prevents hot spots on monotonic values

WILDCARD INDEX INTERNALS:
    → At insert time: traverses entire document tree
    → For each leaf field: creates index entry { path: value }
    → Example: { specs: { wattage: 100 } }
      → Entry: { "specs.wattage": 100 } → document pointer
    → Query { "specs.wattage": { $gt: 50 } }
      → Looks up path "specs.wattage" in index → range scan on values
    → Dynamic: new fields automatically indexed on insert
    → Cost: more write overhead (must index every field in every doc)

MULTIKEY INDEX INTERNALS:
    → At insert time: detects array field
    → Creates ONE index entry PER array element
    → tags: ["a", "b", "c"] → 3 entries: "a"→doc, "b"→doc, "c"→doc
    → Index size: sum of all array lengths across all documents
    → 1M docs × avg 5 tags = 5M index entries (vs 1M for non-array)
    → Multikey flag set on index metadata (isMultiKey: true)
    → $all query: AND of multiple index lookups
    → $in query: OR of multiple index lookups

PARTIAL INDEX INTERNALS:
    → At insert/update time: evaluate partialFilterExpression
    → If document matches filter → add to index
    → If document doesn't match → skip (not in index)
    → Query planner checks: does query filter GUARANTEE
      only matching documents will be needed?
    → If yes → use partial index
    → If no → cannot use (might miss non-indexed documents)
    → Result: smaller B-tree, less RAM, same query performance
    → Trade-off: only queries including the filter condition benefit

SPARSE INDEX INTERNALS:
    → At insert time: check if indexed field EXISTS in document
    → Field present (even if null) → add to index
    → Field missing entirely → skip
    → Connects to Project 8: null vs missing distinction
    → Sparse + unique: allows multiple documents to LACK the field
      without "duplicate null" violations
    → Regular unique index: two documents without the field
      → both have "null" → duplicate key error!
    → Sparse unique index: documents without field are not in index
      → no conflict → both can exist
    → This is critical for optional unique fields (SSN, ISBN, etc.)

HIDDEN INDEX INTERNALS:
    → Index still maintained on writes (B-tree updated normally)
    → Query planner ignores it (pretends it doesn't exist)
    → Zero data change — just a metadata flag
    → Toggle: hideIndex() / unhideIndex()
    → Use case: "would dropping this index hurt anything?"
    → Hide it → monitor for 24 hours → if queries still fast → drop it
    → If queries slow down → unhide immediately → no rebuild needed
    → Redshift: no equivalent (no indexes to hide/unhide)

Internals Comparison: Why MongoDB Has These and Redshift Doesn't

WHY REDSHIFT DOESN'T NEED (most of) THESE INDEX TYPES:

FUNDAMENTAL ARCHITECTURE DIFFERENCE:
    Redshift = ANALYTICAL warehouse
        → Designed for: scan millions of rows, aggregate, report
        → Strength: columnar compression + MPP parallel scan
        → Full table scans are FAST (the intended operation)
        → Point lookups are SLOW (not the intended operation)
        → Sort key + zone maps = sufficient optimization for scan patterns

    MongoDB = OPERATIONAL database
        → Designed for: find specific documents, real-time read/write
        → Strength: diverse index types for diverse access patterns
        → Point lookups are FAST (the intended operation)
        → Full collection scans are SLOW (not the intended operation)
        → Rich index types = necessary for diverse query patterns

WHY EACH TYPE EXISTS IN MONGODB BUT NOT REDSHIFT:

    Text Index:
        Redshift: analytical queries rarely need full-text search
                  (that's Elasticsearch/OpenSearch's job in analytics stack)
        MongoDB:  applications frequently need search (product catalog, content)
                  built-in text index avoids external dependency

    2dsphere:
        Redshift: geo-analytics done at aggregate level (count per region)
                  ST_Distance sufficient for batch processing
        MongoDB:  real-time "nearest store" queries need indexed geo
                  every user request must be <10ms

    Wildcard:
        Redshift: schema is fixed — columns known at CREATE TABLE time
        MongoDB:  schema is flexible — fields vary per document
                  need to index unknown/dynamic fields

    Partial/Sparse:
        Redshift: every row has every column — no "missing" concept
                  no benefit to indexing a subset (no indexes at all)
        MongoDB:  documents vary — some have fields, some don't
                  indexing only relevant documents saves significant RAM

    Multikey:
        Redshift: arrays require normalization (separate table)
                  array searching is a JOIN operation
        MongoDB:  arrays are first-class — need first-class indexing
                  multikey eliminates the need for normalized tables + JOINs

⚠️ COMMON PITFALLS AND FAILURE MODES

MISTAKES REDSHIFT DEVELOPERS MAKE — PROJECT 11 SPECIFIC:

PITFALL 1: "I'll use regex for search since I'm used to LIKE"
    ❌ db.products.find({ description: /comfortable/i })
       → COLLSCAN (unless regex is anchored: /^comfortable/)
       → No ranking, no stemming, no stop words
    ✅ db.products.find({ $text: { $search: "comfortable" } })
       → TEXT index used, ranked results, stemming applied
    WHY: Redshift LIKE is the only option → regex feels natural
         MongoDB has a BETTER option: text index
    EXCEPTION: Use regex for pattern matching (SKU codes, email domains)
               /^PROD-\d{4}$/ → anchored regex CAN use regular index prefix

PITFALL 2: "I created a text index but search returns nothing"
    ❌ Searching for "the" or "a" or "is" → returns nothing
    ✅ Understanding stop words are excluded from text index
    WHY: Text index removes common English words
         "the blue shoe" → indexed as ["blue", "shoe"] (no "the")
         Search for "the" → no matches in index
         This is correct behavior for relevance search
         Redshift LIKE '%the%' would match (but meaninglessly)

PITFALL 3: "I need two text indexes for different search contexts"
    ❌ Text index on {name: "text"} AND text index on {description: "text"}
       → ERROR: only ONE text index per collection
    ✅ One text index including all searchable fields with weights
       → { name: "text", description: "text", tags: "text" }
       → Use weights to control relevance per field
    WHY: MongoDB text index is ONE inverted index per collection
         Multiple fields go into the SAME inverted index
         Weights differentiate field importance

PITFALL 4: "My geo query returns wrong results"
    ❌ coordinates: [40.7484, -73.9857]  → [lat, lng] WRONG
    ✅ coordinates: [-73.9857, 40.7484]  → [lng, lat] CORRECT
    WHY: Same as Project 8 — GeoJSON standard is [longitude, latitude]
         This mistake is SO common it deserves repeating
         Google Maps shows [lat, lng] → muscle memory is wrong for GeoJSON

PITFALL 5: "Wildcard index makes all my queries fast"
    ❌ Creating { "$**": 1 } and expecting optimal performance everywhere
    ✅ Understanding wildcard index limitations:
       → Cannot serve compound queries (multi-field filters)
       → Only one field per query can use wildcard index
       → Much larger than targeted indexes
       → Slower writes (must index every field)
    WHY: Wildcard = safety net for unknown queries
         Compound index = optimal for known queries
         Use wildcard for truly dynamic fields (specs, metadata)
         Use compound for known query patterns (Projects 9-10)

PITFALL 6: "Partial index should work for all queries on indexed fields"
    ❌ Partial index: { status: "active" } on { category: 1 }
       Query: find({ category: "Electronics" })  → partial index NOT used
    ✅ Query MUST include: find({ status: "active", category: "Electronics" })
    WHY: Without status: "active" in the query, MongoDB can't guarantee
         all matching documents are in the partial index
         (archived Electronics products would be missed)
         ALWAYS include the partial filter expression in your query

PITFALL 7: "I created index on array field but compound doesn't work"
    ❌ createIndex({ tags: 1, sizes: 1 }) where both are arrays
       → Error: cannot index parallel arrays
    ✅ Two separate indexes: { tags: 1 } and { sizes: 1 }
       Or: compound with at most ONE array field
       → { tags: 1, category: 1 } ← OK (tags is array, category is scalar)
    WHY: Cartesian product of two arrays would create too many index entries
         tags: ["a","b"] × sizes: [1,2,3] = 6 entries PER document
         MongoDB prevents this to control index size explosion

PITFALL 8: "Sparse and partial are the same thing"
    ❌ Using sparse when you need partial (or vice versa)
    SPARSE:  based on field EXISTENCE only
             → Document has field (even null) → in index
             → Document missing field → not in index
    PARTIAL: based on ARBITRARY filter expression
             → { status: "active" } → only active docs in index
             → { price: { $gt: 0 } } → only priced products in index
             → Much more flexible than sparse
    ✅ Use partial for condition-based filtering
    ✅ Use sparse for existence-based filtering (simpler syntax)
    WHY: Partial (MongoDB 3.2+) largely supersedes sparse
         But sparse still useful for simple "field exists" cases
         And sparse + unique has a specific use case (optional unique fields)

📈 SCALABILITY CONSIDERATIONS

INDEX TYPE SCALABILITY AT DIFFERENT SCALES:

TEXT INDEX:
    → Size: ~1.5x the size of indexed text data
    → 1M products with 500-char descriptions: ~750MB text index
    → RAM intensive — must fit in WiredTiger cache
    → At scale: consider Atlas Search (Project 29) instead
      → Atlas Search uses dedicated search nodes
      → Doesn't compete with database RAM
    → Redshift: LIKE doesn't scale at all (always full scan)

2DSPHERE INDEX:
    → Size: ~60 bytes per document (cell IDs + pointer)
    → 1M stores: ~60MB index — very manageable
    → Performance: O(log n) regardless of collection size
    → Scales beautifully to millions of locations
    → Redshift ST_Distance: O(n) always — degrades linearly

HASHED INDEX:
    → Size: same as regular single-field index (~40-60 bytes per doc)
    → Primary value is at sharding scale (Project 21)
    → Ensures uniform data distribution across shards
    → Prevents "hot shard" problem with monotonic values

WILDCARD INDEX:
    → Size: proportional to total number of fields × documents
    → 1M docs × avg 20 fields = 20M index entries
    → Can be VERY large — monitor carefully
    → Scope with path: { "specs.$**": 1 } to limit size
    → Full wildcard { "$**": 1 } on large collections = dangerous

PARTIAL INDEX:
    → Size: proportional to matching documents only
    → 1M docs, 200K active: index is 80% smaller
    → BEST scalability optimization for skewed data
    → Active/inactive, published/draft patterns benefit enormously
    → Redshift: cannot do this (no indexes, no partial concept)

MULTIKEY INDEX:
    → Size: proportional to total array elements across all documents
    → 1M docs × avg 5 tags = 5M index entries (5x a single-field index)
    → Arrays with 100+ elements per doc → HUGE index
    → Monitor with db.collection.stats().totalIndexSize
    → Consider Bucket Pattern (Project 16) for very large arrays
    → Redshift: normalized table would have same row count (5M rows)

GENERAL SCALING RULE:
    → More specialized index types = more RAM pressure
    → At >100M documents: carefully audit ALL indexes
    → Use $indexStats to find unused indexes → drop them
    → Use partial indexes to reduce index size where possible
    → At extreme scale: Atlas Search replaces text index
                         Sharding distributes all indexes across nodes

📐 DATA MODELING COMPARISON — Index Types Impact

HOW INDEX TYPES INFLUENCE DOCUMENT DESIGN:

REDSHIFT MODEL (normalized, type-restricted):
    products        (id, name, description, price, category, status)
    product_tags    (product_id, tag)           ← normalized array
    product_sizes   (product_id, size)          ← normalized array
    product_specs   (product_id, key, value)    ← EAV pattern for dynamic attrs
    stores          (id, name, lat, lng, city)  ← separate lat/lng columns
    
    To search: LIKE on products.description (full scan)
    To find by tag: JOIN product_tags (extra table)
    To find by spec: JOIN product_specs (EAV, terrible performance)
    To find by location: ST_Distance on lat/lng (full scan)
    To filter active: WHERE status = 'active' (full scan)
    
    TOTAL TABLES: 5 (for what MongoDB does in 2 collections)
    TOTAL JOINs needed: 3+ per complex query

MONGODB MODEL (denormalized, index-rich):
    products_full   (all fields embedded: tags[], sizes[], specs{})
    stores          (location as GeoJSON)
    
    To search: $text with text index (indexed, ranked)
    To find by tag: multikey index on tags (no JOIN)
    To find by spec: wildcard index on specs (no separate table)
    To find by location: 2dsphere index on location (indexed)
    To filter active: partial index (index only active docs)
    
    TOTAL COLLECTIONS: 2
    TOTAL $lookups needed: 0 for these queries

KEY INSIGHT:
    MongoDB's rich index types ENABLE the denormalized document model
    Without multikey indexes → arrays would be unsearchable → must normalize
    Without wildcard indexes → dynamic fields would be unsearchable → must EAV
    Without text indexes → text search impossible → must external service
    Without 2dsphere → geo queries are full scans → must external service
    
    The index types and document model are SYMBIOTIC
    Each enables the other
    Redshift's lack of index types FORCES normalization and external tools

# ─── HELPER: Explain analyzer ─────────────────────────

def analyze(collection, filter_doc, label, sort_doc=None,
            projection=None, limit_val=None):
    """Quick explain analysis for index type demonstrations."""
    cursor = collection.find(filter_doc, projection or {})
    if sort_doc:
        cursor = cursor.sort(list(sort_doc.items()))
    if limit_val:
        cursor = cursor.limit(limit_val)

    explain = cursor.explain()
    stats = explain["executionStats"]

    # Walk plan tree for index info
    node = explain["queryPlanner"]["winningPlan"]
    stages = []
    index_name = None
    is_multikey = None
    while node:
        stages.append(node.get("stage", "?"))
        if node.get("indexName"):
            index_name = node["indexName"]
        if node.get("isMultiKey") is not None:
            is_multikey = node["isMultiKey"]
        node = node.get("inputStage")

    is_covered = (stats["totalDocsExamined"] == 0 and
                  stats["totalKeysExamined"] > 0)

    print(f"\n  [{label}]")
    print(f"    Stages:    {' → '.join(stages)}")
    print(f"    Index:     {index_name or 'NONE (COLLSCAN)'}")
    if is_multikey is not None:
        print(f"    Multikey:  {is_multikey}")
    print(f"    Keys:      {stats['totalKeysExamined']}")
    print(f"    Docs exam: {stats['totalDocsExamined']}")
    print(f"    Returned:  {stats['nReturned']}")
    print(f"    Time:      {stats['executionTimeMillis']}ms")
    if is_covered:
        print(f"    🟢 COVERED QUERY")

    return stats
